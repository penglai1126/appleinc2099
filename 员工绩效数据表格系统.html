/**
 * 导出标准CSV模板
 * @returns {void}
 */
function exportCSVTemplate() {
    const headers = [
        '员工ID', '姓名', '部门', '工作进度', '任务数量',
        '完成数量', '质量评分', '工作时长', '绩效系数', '备注'
    ];
    
    const templateData = [
        headers,
        ['EMP001', '示例姓名', '技术部', '85%', '10', '8', '4.5', '8.5', '1.2', '请勿修改表头']
    ];
    
    const csvContent = templateData.map(row => 
        row.map(field => `"${field}"`).join(',')
    ).join('\n');
    
    const blob = new Blob([csvContent], { type: 'text/csv;charset=utf-8;' });
    const link = document.createElement('a');
    link.href = URL.createObjectURL(blob);
    link.download = `员工绩效模板_${new Date().toISOString().split('T')[0]}.csv`;
    link.click();
    URL.revokeObjectURL(link.href);
}

2. 批量CSV导入系统

/**
 * 批量导入CSV文件并按文件名排序
 * @param {FileList} files - 文件列表
 * @returns {Promise<Array>} 合并后的数据
 */
async function importCSVFiles(files) {
    const fileArray = Array.from(files)
        .filter(file => file.name.endsWith('.csv'))
        .sort((a, b) => a.name.localeCompare(b.name, 'zh-CN'));
    
    const allData = [];
    
    for (const file of fileArray) {
        try {
            const text = await readFileAsText(file);
            const data = parseCSV(text);
            allData.push({
                filename: file.name,
                data: data.slice(1) // 跳过表头
            });
        } catch (error) {
            console.error(`文件 ${file.name} 处理失败:`, error);
        }
    }
    
    return allData.flatMap(item => item.data);
}

/**
 * 解析CSV内容
 * @param {string} csvText - CSV文本内容
 * @returns {Array<Array<string>>} 解析后的数据
 */
function parseCSV(csvText) {
    const rows = csvText.split('\n').filter(row => row.trim());
    return rows.map(row => {
        const result = [];
        let current = '';
        let inQuotes = false;
        
        for (let char of row) {
            if (char === '"') {
                inQuotes = !inQuotes;
            } else if (char === ',' && !inQuotes) {
                result.push(current);
                current = '';
            } else {
                current += char;
            }
        }
        result.push(current);
        return result;
    });
}

3. 核心计算功能按钮（5个）

按钮1: 绩效等级计算

function calculatePerformanceLevel(data) {
    return data.map(employee => {
        const progress = parseFloat(employee[3]);
        const quality = parseFloat(employee[6]);
        const hours = parseFloat(employee[7]);
        
        // 逻辑判断组合公式应用
        let level;
        if (progress >= 90 && quality >= 4.5) {
            level = "卓越";
        } else if (progress >= 80 || quality >= 4.0) {
            level = "优秀";
        } else if (progress >= 70 && hours >= 7) {
            level = "良好";
        } else {
            level = "待提升";
        }
        
        return [...employee, level];
    });
}

按钮2: 综合得分计算

function calculateCompositeScore(data) {
    return data.map(employee => {
        const progress = parseFloat(employee[3]);
        const taskCount = parseFloat(employee[4]);
        const completed = parseFloat(employee[5]);
        const quality = parseFloat(employee[6]);
        const coefficient = parseFloat(employee[8]);
        
        // 权重评分组合公式
        const weights = [0.3, 0.2, 0.25, 0.25]; // 进度、完成率、质量、任务数量权重
        const completionRate = completed / taskCount * 100;
        
        const scores = [
            progress,
            completionRate,
            quality * 20, // 转换为百分制
            taskCount * 2 // 任务数量加权
        ];
        
        const weightedScore = scores.reduce((sum, score, index) => 
            sum + score * weights[index], 0
        ) * coefficient;
        
        return [...employee, completionRate.toFixed(1) + '%', weightedScore.toFixed(2)];
    });
}

按钮3: 部门统计分析

function departmentStatistics(data) {
    // 条件统计组合公式应用
    const departments = [...new Set(data.map(row => row[2]))];
    
    return departments.map(dept => {
        const deptData = data.filter(row => row[2] === dept);
        
        const stats = {
            department: dept,
            employeeCount: deptData.length,
            avgProgress: deptData.reduce((sum, row) => 
                sum + parseFloat(row[3]), 0) / deptData.length,
            avgQuality: deptData.reduce((sum, row) => 
                sum + parseFloat(row[6]), 0) / deptData.length,
            totalHours: deptData.reduce((sum, row) => 
                sum + parseFloat(row[7]), 0)
        };
        
        return stats;
    });
}

按钮4: 排名分析

function calculateRankings(data) {
    // 多条件排名组合公式
    return data.map((employee, index, array) => {
        const currentScore = parseFloat(employee[employee.length - 1]); // 综合得分
        
        const rank = array.filter(emp => 
            parseFloat(emp[emp.length - 1]) > currentScore
        ).length + 1;
        
        const departmentRank = array
            .filter(emp => emp[2] === employee[2])
            .filter(emp => parseFloat(emp[emp.length - 1]) > currentScore)
            .length + 1;
        
        return [...employee, rank, departmentRank];
    });
}

按钮5: 数据质量检查

function dataQualityCheck(data) {
    // 数据验证组合公式
    return data.map(employee => {
        const errors = [];
        
        // 员工ID验证
        if (!/^EMP\d{3}$/.test(employee[0])) {
            errors.push('员工ID格式错误');
        }
        
        // 进度验证
        const progress = parseFloat(employee[3]);
        if (isNaN(progress) || progress < 0 || progress > 100) {
            errors.push('工作进度范围错误');
        }
        
        // 质量评分验证
        const quality = parseFloat(employee[6]);
        if (isNaN(quality) || quality < 1 || quality > 5) {
            errors.push('质量评分范围错误');
        }
        
        return {
            ...employee,
            dataQuality: errors.length === 0 ? '合格' : '问题:' + errors.join(',')
        };
    });
}

实时数据验证

/**
 * 实时数据验证系统
 */
class DataValidator {
    static validateEmployeeData(row) {
        const validations = [
            { index: 0, validator: this.validateEmployeeID },
            { index: 3, validator: this.validateProgress },
            { index: 6, validator: this.validateQuality },
            { index: 7, validator: this.validateWorkHours }
        ];
        
        return validations.every(validation => 
            validation.validator(row[validation.index])
        );
    }
    
    static validateEmployeeID(id) {
        return /^EMP\d{3}$/.test(id);
    }
    
    static validateProgress(progress) {
        const value = parseFloat(progress);
        return !isNaN(value) && value >= 0 && value <= 100;
    }
}

/**
 * 大数据集分块处理
 */
async function processLargeDataset(files, chunkSize = 1000) {
    const results = [];
    
    for (const file of files) {
        const text = await readFileAsText(file);
        const rows = parseCSV(text).slice(1); // 跳过表头
        
        // 分块处理避免内存溢出
        for (let i = 0; i < rows.length; i += chunkSize) {
            const chunk = rows.slice(i, i + chunkSize);
            const processedChunk = processDataChunk(chunk);
            results.push(...processedChunk);
            
            // 更新进度显示
            updateProgress((i / rows.length) * 100);
        }
    }
    
    return results;
}

        /**
         * 处理文件导入
         */
        async function handleFileImport(files) {
            if (!files || files.length === 0) return;
            
            const csvFiles = Array.from(files)
                .filter(file => file.name.toLowerCase().endsWith('.csv'))
                .sort((a, b) => a.name.localeCompare(b.name, 'zh-CN'));
            
            if (csvFiles.length === 0) {
                showStatus('❌ 请选择CSV格式的文件', 'error');
                return;
            }
            
            updateProgress(0, `开始处理 ${csvFiles.length} 个文件...`);
            
            // 显示文件列表
            const fileListDiv = document.getElementById('fileList');
            fileListDiv.innerHTML = `<h4>已选择文件 (${csvFiles.length}个):</h4>` + 
                csvFiles.map((file, index) => 
                    `<div style="margin: 4px 0; color: var(--text-secondary);">${index + 1}. ${file.name}</div>`
                ).join('');
            
            try {
                const allData = [];
                let processedFiles = 0;
                
                for (const file of csvFiles) {
                    const text = await readFileAsText(file);
                    const data = parseCSV(text);
                    
                    if (data.length > 1) {
                        // 跳过表头，只取数据行
                        allData.push(...data.slice(1));
                    }
                    
                    processedFiles++;
                    updateProgress((processedFiles / csvFiles.length) * 100, 
                                 `处理中: ${processedFiles}/${csvFiles.length} - ${file.name}`);
                }
                
                currentData = allData;
                processedData = [...allData];
                
                // 更新统计信息
                updateStatistics();
                
                // 显示数据表格
                displayDataTable(allData);
                
                updateProgress(100, `✅ 成功导入 ${allData.length} 条员工数据`);
                showStatus(`✅ 数据导入完成！共处理 ${csvFiles.length} 个文件，导入 ${allData.length} 条记录`, 'success');
                
            } catch (error) {
                console.error('文件处理错误:', error);
                showStatus(`❌ 文件处理失败: ${error.message}`, 'error');
                updateProgress(0, '');
            }
        }

        /**
         * 读取文件为文本
         */
        function readFileAsText(file) {
            return new Promise((resolve, reject) => {
                const reader = new FileReader();
                reader.onload = e => resolve(e.target.result);
                reader.onerror = reject;
                reader.readAsText(file, 'UTF-8');
            });
        }

        /**
         * 解析CSV内容
         */
        function parseCSV(csvText) {
            const rows = csvText.split('\n').filter(row => row.trim());
            return rows.map(row => {
                const result = [];
                let current = '';
                let inQuotes = false;
                
                for (let char of row) {
                    if (char === '"') {
                        inQuotes = !inQuotes;
                    } else if (char === ',' && !inQuotes) {
                        result.push(current);
                        current = '';
                    } else {
                        current += char;
                    }
                }
                result.push(current);
                return result.map(field => field.trim().replace(/^"|"$/g, ''));
            });
        }

        // 核心计算函数
        function calculatePerformanceLevel(data) {
            return data.map(employee => {
                const progress = parseFloat(employee[3]) || 0;
                const quality = parseFloat(employee[6]) || 0;
                const hours = parseFloat(employee[7]) || 0;
                
                let level;
                if (progress >= 90 && quality >= 4.5) {
                    level = "卓越";
                } else if (progress >= 80 || quality >= 4.0) {
                    level = "优秀";
                } else if (progress >= 70 && hours >= 7) {
                    level = "良好";
                } else {
                    level = "待提升";
                }
                
                return [...employee, level];
            });
        }

        function calculateCompositeScore(data) {
            return data.map(employee => {
                const progress = parseFloat(employee[3]) || 0;
                const taskCount = parseFloat(employee[4]) || 0;
                const completed = parseFloat(employee[5]) || 0;
                const quality = parseFloat(employee[6]) || 0;
                const coefficient = parseFloat(employee[8]) || 1.0;
                
                const weights = [0.3, 0.2, 0.25, 0.25];
                const completionRate = taskCount > 0 ? (completed / taskCount) * 100 : 0;
                
                const scores = [
                    progress,
                    completionRate,
                    quality * 20,
                    Math.min(taskCount * 2, 100)
                ];
                
                const weightedScore = scores.reduce((sum, score, index) => 
                    sum + score * weights[index], 0
                ) * coefficient;
                
                return [...employee, completionRate.toFixed(1) + '%', weightedScore.toFixed(1)];
            });
        }

        function departmentStatistics(data) {
            const departments = [...new Set(data.map(row => row[2]))];
            
            return departments.map(dept => {
                const deptData = data.filter(row => row[2] === dept);
                const validDeptData = deptData.filter(row => 
                    !isNaN(parseFloat(row[3])) && !isNaN(parseFloat(row[6])) && !isNaN(parseFloat(row[7]))
                );
                
                if (validDeptData.length === 0) {
                    return {
                        department: dept,
                        employeeCount: deptData.length,
                        avgProgress: 0,
                        avgQuality: 0,
                        totalHours: 0
                    };
                }
                
                return {
                    department: dept,
                    employeeCount: deptData.length,
                    avgProgress: validDeptData.reduce((sum, row) => sum + parseFloat(row[3]), 0) / validDeptData.length,
                    avgQuality: validDeptData.reduce((sum, row) => sum + parseFloat(row[6]), 0) / validDeptData.length,
                    totalHours: validDeptData.reduce((sum, row) => sum + parseFloat(row[7]), 0)
                };
            });
        }

        function calculateRankings(data) {
            // 假设综合得分在第11列（索引10）
            return data.map((employee, index, array) => {
                const currentScore = parseFloat(employee[11]) || 0;
                
                const totalRank = array.filter(emp => 
                    (parseFloat(emp[11]) || 0) > currentScore
                ).length + 1;
                
                const departmentRank = array
                    .filter(emp => emp[2] === employee[2])
                    .filter(emp => (parseFloat(emp[11]) || 0) > currentScore)
                    .length + 1;
                
                return [...employee, totalRank, departmentRank];
            });
        }

        function dataQualityCheck(data) {
            return data.map(employee => {
                const errors = [];
                
                // 员工ID验证
                if (!/^EMP\d{3}$/.test(employee[0])) {
                    errors.push('员工ID格式错误');
                }
                
                // 进度验证
                const progress = parseFloat(employee[3]);
                if (isNaN(progress) || progress < 0 || progress > 100) {
                    errors.push('工作进度范围错误');
                }
                
                // 质量评分验证
                const quality = parseFloat(employee[6]);
                if (isNaN(quality) || quality < 1 || quality > 5) {
                    errors.push('质量评分范围错误');
                }
                
                return {
                    ...employee,
                    dataQuality: errors.length === 0 ? '合格' : '问题:' + errors.join(',')
                };
            });
        }

        /**
         * 导出结果
         */
        function exportResults(format) {
            if (processedData.length === 0) {
                showStatus('❌ 没有可导出的数据', 'error');
                return;
            }
            
            const timestamp = new Date().toISOString().split('T')[0];
            
            if (format === 'xlsx') {
                // 使用SheetJS导出XLSX
                const ws = XLSX.utils.aoa_to_sheet(processedData);
                const wb = XLSX.utils.book_new();
                XLSX.utils.book_append_sheet(wb, ws, '员工绩效数据');
                XLSX.writeFile(wb, `员工绩效数据_${timestamp}.xlsx`);
                showStatus('💾 XLSX文件导出成功！', 'success');
            } else if (format === 'csv') {
                const csvContent = processedData.map(row => 
                    row.map(field => `"${field}"`).join(',')
                ).join('\n');
                
                const blob = new Blob([csvContent], { type: 'text/csv;charset=utf-8;' });
                const link = document.createElement('a');
                link.href = URL.createObjectURL(blob);
                link.download = `员工绩效数据_${timestamp}.csv`;
                link.click();
                URL.revokeObjectURL(link.href);
                showStatus('📋 CSV文件导出成功！', 'success');
            }
        }
